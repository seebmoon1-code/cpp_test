void greetUser(std::wstring userName) {
    // تنظیمات locale برای اطمینان از خروجی UTF-8

#include <iostream>
#include <string>

// تعریف تابع
// این تابع یک رشته (string) به عنوان نام می‌گیرد
// و یک پیام سلام به کاربر در خروجی چاپ می‌کند.
void greetUser(std::string userName) {
    std::cout << "سلام، " << userName << "! خوش آمدید." << std::endl;
}

// تابع اصلی (main function) برنامه
int main() {
    // 1. یک متغیر برای ذخیره نام کاربر تعریف می‌کنیم
    std::string name;

    // 2. از کاربر می‌خواهیم نامش را وارد کند
    std::cout << "لطفاً نام خود را وارد کنید: ";
    
    // 3. نام وارد شده توسط کاربر را می‌خوانیم
    // از getline استفاده می‌کنیم تا بتوانیم نام‌هایی با فاصله (مانند "علی محمدی") را هم بخوانیم.
    std::getline(std::cin, name);

    // 4. تابع greetUser را با نام وارد شده فراخوانی می‌کنیم
    greetUser(name);

    return 0; // پایان موفقیت‌آمیز برنامه
}
